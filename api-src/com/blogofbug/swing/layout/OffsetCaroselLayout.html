016 package com.blogofbug.swing.layout;
017 
018 import com.blogofbug.swing.components.RichComponent;
019 
020 import java.awt.Component;
021 import java.awt.Container;
022 import java.awt.Dimension;
023 import java.awt.Insets;
024 import java.awt.Point;
025 
026 /**
027  * Offsets a normal carousel layout be a fixed amount allowing it be "moved" around the screen.
028  *
029  * @author nigel
030  */
031 public class OffsetCaroselLayout extends CaroselLayout {
032 
033     /**
034      * Creates a new instance of OffsetCaroselLayout
035      *
036      * @param forContainer The container to associate the layout with.
037      */
038     public OffsetCaroselLayout(Container forContainer) {
039         super(forContainer);
040     }
041 
042     //Checks to see if the component should be hidden
043 
044     /**
045      * Overrides the normal layout method to determine if the object is offscreen and can therefore
046      * be ignored (giving a performance gain). In addition, it will adjust the alpha of the component
047      * based on its distance from 3'oclock (or 15:00 if you use a 24-hour compass)
048      *
049      * @param comp  The component to consider
050      * @param angle Its position on the carousel.
051      * @param scale The scale (applied to size) of the image, that is, how far from the observer is it
052      * @return True if it should be hidden, false if it should not
053      */
054     protected boolean shouldHide(Component comp, double angle, double scale) {
055         //Quick test where it's visible
056 
057         double cos = Math.cos(angle);
058         double sin = Math.sin(angle);
059         if ((sin > -0.5) && (sin < 1) && (cos < 0.3)) {
060             //We are not going to hide it, but we will set the alpha
061             if (scale > 1.00) {
062                 scale = Math.abs(scale - 1.0);
063                 scale *= 3.0;
064                 scale = 1.0 - scale;
065             } else {
066                 scale = 1.0;
067 
068             }
069             if (comp instanceof RichComponent) {
070                 ((RichComponent) comp).setAlpha((float) scale);
071             }
072             return false;
073         }
074 
075         return true;
076     }
077 
078     /**
079      * Determines how "wide" the carousel should be drawn based on the side of the container the layout is laying out
080      *
081      * @param target          The container the radius should be calculated for
082      * @param insets          Any insets of the container
083      * @param width           The width of the container
084      * @param height          The height of the container
085      * @param widestComponent The widest component
086      * @return The size (in a bounding box) of the carousel
087      */
088     protected Dimension getCarouselRadius(Container target, Insets insets, int width, int height, int widestComponent) {
089         width = target.getSize().width - (insets.left + insets.right + (widestComponent / 2));
090 
091         height = (int) (height / 2.5);
092 
093         return new Dimension(width, height);
094     }
095 
096     /**
097      * Determines the center of the carousel based on the dimensions of the container
098      *
099      * @param insets The container insets
100      * @param width  Width of the container
101      * @param height The height of the container
102      * @param widest width of the container
103      * @return A point representing the new center
104      */
105     protected Point calculateCenter(Insets insets, int width, int height, int widest) {
106         return new Point(0, ((height - insets.bottom) / 2) + widest / 3);
107     }
108 
109     /**
110      * Over-rides the normal setFrontMostComponent to move the selected component to 3 o'clock instead of 6 o'clock
111      *
112      * @param component The component to move to the 'front'
113      */
114     public void setFrontMostComponent(Component component) {
115         setTarget(-getPosition(component).getTargetAngle() + (Math.PI / 2));
116     }
117 
118 }
