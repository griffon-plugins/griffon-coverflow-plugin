016 package com.blogofbug.swing.borders;
017 
018 import javax.swing.border.AbstractBorder;
019 import java.awt.Component;
020 import java.awt.Graphics;
021 import java.awt.Graphics2D;
022 import java.awt.Insets;
023 import java.awt.image.BufferedImage;
024 import java.net.URL;
025 
026 /**
027  * ImageBorder takes an image and breaks out the corners and the top, left, right, and bottom
028  * borders stretching them to fill the space around which the border is drawn
029  *
030  * @author nigel
031  */
032 public class ImageBorder extends AbstractBorder {
033     protected AbstractImageBorder borderRenderer;
034     boolean paintBorder = true;
035 
036     /**
037      * Creates a new ImageBorder using the supplied image and the insets
038      *
039      * @param borderImage The image to be used as the border
040      * @param imageInsets The insets around the edge of the image that allow the cookie-cut-and-stretch of the image
041      *                    around the edge of the border
042      */
043     public ImageBorder(BufferedImage borderImage, Insets imageInsets) {
044         borderRenderer = new AbstractImageBorder(borderImage, imageInsets);
045     }
046 
047 
048     /**
049      * Creates a new ImageBofder loading the image from the supplied URL
050      *
051      * @param imageURL    The location of the image to use
052      * @param imageInsets The insets around the edge of the image that allow the cookie-cut-and-stretch of the image
053      *                    around the edge of the border
054      */
055     public ImageBorder(URL imageURL, Insets imageInsets) {
056         borderRenderer = new AbstractImageBorder(imageURL, imageInsets);
057     }
058 
059     /**
060      * Paints the border around the specified component
061      *
062      * @param c      The component to paint the border on
063      * @param g      The graphics context
064      * @param x      The x offset
065      * @param y      The y offset
066      * @param width  The width
067      * @param height The height
068      */
069     public void paintBorder(Component c, Graphics g, int x, int y, int width, int height) {
070         if (!paintBorder) {
071             return;
072         }
073         borderRenderer.paintBorder(c.getWidth(), c.getHeight(), g, x, y, width, height);
074     }
075 
076     /**
077      * Controls wether or not the border is actually painted or not.
078      *
079      * @param paintBorder If false then will not draw the border. Useful if the border is being used to show a selected item
080      */
081     public void setPaintBorder(boolean paintBorder) {
082         this.paintBorder = paintBorder;
083     }
084 
085     /**
086      * Gets the insets of the image back (subtracting from the component size would give you the renderable
087      * area
088      *
089      * @param c The component to which the border will be applied
090      * @return The insets of the border
091      */
092     public Insets getBorderInsets(Component c) {
093         return borderRenderer.getImageInsets();
094     }
095 
096     /**
097      * Gets the insets of the image and returns in the in the supplied Insets instance
098      *
099      * @param c The component to which the border will be applied
100      * @param i A pre-created insets object
101      * @return The insets of the border
102      */
103     public Insets getBorderInsets(Component c, Insets i) {
104         Insets imageInsets = borderRenderer.getImageInsets();
105         i.top = imageInsets.top;
106         i.bottom = imageInsets.bottom;
107         i.left = imageInsets.left;
108         i.right = imageInsets.right;
109         return i;
110     }
111 
112     public void paintCenter(Graphics2D g2, Component c) {
113         if (paintBorder) {
114             borderRenderer.paintCenter(g2, c.getWidth(), c.getHeight());
115         }
116     }
