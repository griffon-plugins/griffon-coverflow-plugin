016 package com.blogofbug.swing.borders;
017 
018 import com.blogofbug.utility.ImageUtilities;
019 
020 import java.awt.Graphics;
021 import java.awt.Graphics2D;
022 import java.awt.Insets;
023 import java.awt.image.BufferedImage;
024 import java.net.URL;
025 
026 /**
027  * @author nigel
028  */
029 public class AbstractImageBorder {
030 
031     /**
032      * Creates a new ImageBofder loading the image from the supplied URL
033      *
034      * @param imageURL    The location of the image to use
035      * @param imageInsets The insets around the edge of the image that allow the cookie-cut-and-stretch of the image
036      *                    around the edge of the border
037      */
038     public AbstractImageBorder(URL imageURL, Insets imageInsets) {
039         this.imageInsets = imageInsets;
040         borderImage = ImageUtilities.loadCompatibleImage(imageURL.toString());
041     }
042 
043     /**
044      * Creates a new ImageBorder using the supplied image and the insets
045      *
046      * @param borderImage The image to be used as the border
047      * @param imageInsets The insets around the edge of the image that allow the cookie-cut-and-stretch of the image
048      *                    around the edge of the border
049      */
050     public AbstractImageBorder(BufferedImage borderImage, Insets imageInsets) {
051         this.borderImage = borderImage;
052         this.imageInsets = imageInsets;
053     }
054 
055 
056     protected BufferedImage borderImage;
057 
058 
059     protected Insets imageInsets;
060 
061 
062     /**
063      * Paints the border around the specified component
064      *
065      * @param c      The component to paint the border on
066      * @param g      The graphics context
067      * @param x      The x offset
068      * @param y      The y offset
069      * @param width  The width
070      * @param height The height
071      */
072     public void paintBorder(int compWidth, int compHeight, Graphics g, int x, int y, int width, int height) {
073 
074         Graphics2D g2 = (Graphics2D) g;
075         int imageWidth = borderImage.getWidth();
076         int imageHeight = borderImage.getHeight();
077 
078         //Top-left corner
079         drawSlice(g2, 0, 0, imageInsets.left, imageInsets.top, 0, 0);
080         //Top-right corner
081         drawSlice(g2, imageWidth - imageInsets.right, 0, imageInsets.right, imageInsets.bottom,
082             compWidth - imageInsets.right, 0);
083         //Bottom-left corner
084         drawSlice(g2, 0, imageHeight - imageInsets.bottom, imageInsets.left, imageInsets.bottom, 0, compHeight - imageInsets.bottom);
085 
086         //Bottom-right corner
087         drawSlice(g2, imageWidth - imageInsets.right, imageHeight - imageInsets.bottom, imageInsets.left, imageInsets.bottom, compWidth - imageInsets.right, compHeight - imageInsets.bottom);
088 
089         //Draw left side
090         g2.drawImage(borderImage, 0, imageInsets.top, imageInsets.left, compHeight - imageInsets.bottom,
091             0, imageInsets.top, imageInsets.left, imageHeight - imageInsets.bottom, null);
092 
093         //Draw right side
094         g2.drawImage(borderImage, compWidth - imageInsets.right, imageInsets.top + 6, compWidth, compHeight - imageInsets.bottom,
095             imageWidth - imageInsets.right, imageInsets.top, imageWidth, imageHeight - imageInsets.bottom, null);
096 
097         //Draw top side
098         g2.drawImage(borderImage, imageInsets.left, 0, compWidth - imageInsets.left, imageInsets.top,
099             imageInsets.left, 0, imageWidth - imageInsets.right, imageInsets.top, null);
100 
101         //Draw bottom side
102         g2.drawImage(borderImage, imageInsets.left, compHeight - imageInsets.bottom, compWidth - imageInsets.left, compHeight,
103             imageInsets.left, imageHeight - imageInsets.bottom, imageWidth - imageInsets.right, imageHeight, null);
104 
105     }
106 
107 
108     /**
109      * Sets the insets around the edge of the image to be used to cookie cut the image into a border
110      *
111      * @param insets The edges of the image
112      */
113     public void setInsets(Insets insets) {
114         this.imageInsets = insets;
115     }
116 
117     public Insets getImageInsets() {
118         return (Insets) imageInsets.clone();
119     }
120 
121     /**
122      * Paints a stretched version of the center of the image (as the border is drawn
123      * first, then the component paints itself) so that the component can use it in
124      * its own paint if the border lends itself to having a centre area over-painted
125      *
126      * @param g2 The graphics context
127      * @param c  The component
128      */
129     public void paintCenter(Graphics2D g2, int compWidth, int compHeight) {
130         int imageWidth = borderImage.getWidth();
131         int imageHeight = borderImage.getHeight();
132 
133         //draw center
134         g2.drawImage(borderImage, imageInsets.left, imageInsets.top, compWidth - imageInsets.right, compHeight - imageInsets.bottom,
135             imageInsets.left, imageInsets.top, imageWidth - imageInsets.right, imageHeight - imageInsets.bottom, null);
136     }
137 
138     /**
139      * Draws a slicde from the specified image onto the graphics area
140      *
141      * @param g2          The graphics context to draw into
142      * @param sliceX      The x-cordinate of the slice
143      * @param sliceY      The y-cordinate of the slice
144      * @param sliceWidth  The width of the slice
145      * @param sliceHeight The height of the slice
146      * @param destX       The target location of the drawn slice
147      * @param destY       The target location of the drawn slice
148      */
149     private void drawSlice(Graphics2D g2, int sliceX, int sliceY, int sliceWidth, int sliceHeight, int destX, int destY) {
150         g2.drawImage(borderImage, destX, destY, destX + sliceWidth, destY + sliceHeight,
151             sliceX, sliceY, sliceX + sliceWidth, sliceY + sliceHeight, null);
152     }
