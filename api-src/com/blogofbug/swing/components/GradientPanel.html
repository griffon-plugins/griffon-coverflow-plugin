019 import javax.swing.JPanel;
020 import java.awt.Color;
021 import java.awt.GradientPaint;
022 import java.awt.Graphics;
023 import java.awt.Graphics2D;
024 import java.awt.event.ComponentEvent;
025 import java.awt.event.ComponentListener;
026 import java.awt.image.BufferedImage;
027 
028 /**
029  * Container that draws (in an optimized way) a gradient in the background
030  *
031  * @author bug
032  *         <p/>
033  *         Really trivial panel to draw the nice graduated background.
034  */
035 public class GradientPanel extends JPanel implements ComponentListener {
036     /**
037      * Gradient start colour
038      */
039     protected Color start;
040     /**
041      * Gradient end color
042      */
043     protected Color end;
044     /**
045      * Gradient painter
046      */
047     protected GradientPaint gp = null;
048     /**
049      * A pre-rendered gradient in an image
050      */
051     protected BufferedImage cache = null;
052 
053 
054     /**
055      * Set the background to a single color
056      *
057      * @param color The color for a solid background
058      */
059     public void setBackground(Color color) {
060         this.start = color;
061         this.end = color;
062         super.setBackground(color);
063     }
064 
065     /**
066      * Sets two background colors for a gradient
067      *
068      * @param start Top (first) color
069      * @param end   Bottom (final) color
070      */
071     public void setBackground(Color start, Color end) {
072         this.start = start;
073         this.end = end;
074         makeGradient();
075     }
076 
077     /**
078      * paints the gradient.
079      *
080      * @param graphics The graphics context
081      */
082     public void paintComponent(Graphics graphics) {
083         if (start == end) {
084             super.paintComponent(graphics);
085             return;
086         }
087 
088         Graphics2D g2 = (Graphics2D) graphics;
089         
090         /*
091         //Thanks Romain Guy
092         if (cache == null || cache.getHeight() != getHeight()) {
093             cache = new BufferedImage(2, getHeight(),
094                 BufferedImage.TYPE_INT_RGB);
095             Graphics2D g2d = cache.createGraphics();
096             
097             GradientPaint paint = new GradientPaint(0, 0, start,
098                 0, getHeight(), end);
099             g2d.setPaint(paint);
100             g2d.fillRect(0, 0, 2, getHeight());
101             g2d.dispose();
102         }
103         g2.setPaint(new TexturePaint(cache, new Rectangle(0, 0, 1, getHeight())));
104         g2.fillRect(0, 0, getWidth(), getHeight());
105         
106         //g2.drawImage(cache, 0, 0, getWidth(), getHeight(), null);        
107         */
108         gp = new GradientPaint((float) (getWidth() / 2), (float) getY(), start, (float) (getWidth() / 2), (float) getHeight(), end, false);
109 
110         g2.setPaint(gp);
111         g2.fillRect(0, 0, getWidth(), getHeight());
112         super.paintChildren(graphics);
113     }
114 
115     /**
116      * Pre-renders the gradient
117      */
118     private void makeGradient() {
119         gp = new GradientPaint((float) (getWidth() / 2), (float) getY(), start, (float) (getWidth() / 2), (float) getHeight(), end, false);
120     }
121 
122     /**
123      * Recalculates the gradient when it's resized
124      *
125      * @param componentEvent The event object
126      */
127     public void componentResized(ComponentEvent componentEvent) {
128         makeGradient();
129     }
130 
131     /**
132      * Ignored
133      *
134      * @param componentEvent The component event
135      */
136     public void componentShown(ComponentEvent componentEvent) {
137         makeGradient();
138     }

141      * Not used *
142      *
143      * @param componentEvent The event
144      */
145     public void componentMoved(ComponentEvent componentEvent) {
146     }

149      * Not used *
150      *
151      * @param componentEvent The event
152      */
153     public void componentHidden(ComponentEvent componentEvent) {
154     }
