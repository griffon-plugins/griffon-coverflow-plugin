16 package com.blogofbug.swing.components;
17 
18 import javax.swing.AbstractAction;
19 import javax.swing.Action;
20 import java.net.URL;
21 
22 /**
23  * An action for the CarouselMenuAction. May be later refactored into a more useful set of fields or removed completely.
24  *
25  * @author nigel
26  */
27 public abstract class AbstractCarouselMenuAction extends AbstractAction {
28     /**
29      * A very high resolution (recommend at leat 128x128) image to associate with the action
30      */
31     public static final String ACTION_IMAGE_URL = "actionImageURL";
32 
33     /**
34      * Creates a new instance of AbstractCarouselMenuAction
35      *
36      * @param image The image
37      * @param label The text
38      */
39     public AbstractCarouselMenuAction(URL image, String label) {
40         super();
41         this.putValue(ACTION_IMAGE_URL, image);
42         this.putValue(Action.SHORT_DESCRIPTION, label);
43     }

45 }
