16 package com.blogofbug.swing.components;
17 
18 import javax.swing.ImageIcon;
19 import javax.swing.JLabel;
20 import java.awt.Dimension;
21 import java.awt.Graphics;
22 import java.awt.Graphics2D;
23 import java.awt.Image;
24 import java.awt.RenderingHints;
25 import java.awt.image.ImageObserver;
26 
27 /**
28  * A simple component that scales an image to fit inside the size of the component.
29  *
30  * @author nigel
31  */
32 public class ImageLabel extends JLabel {
33     /**
34      * The image that ends up getting scaled
35      */
36     protected ImageIcon imageIcon = null;
37 
38     /**
39      * Creates a new instance of ImageLabel. The prefered width and height will
40      * be set to the dimensions of the image
41      *
42      * @param icon The image to display
43      */
44     public ImageLabel(ImageIcon icon) {
45         super(icon);
46         this.imageIcon = icon;
47         setPreferredSize(new Dimension(icon.getIconWidth(), icon.getIconHeight()));
48     }
49 
50     /**
51      * Creates a new instance of ImageLabel, setting the preferred rendering size to
52      * the supplied dimensions
53      *
54      * @param icon   The image to place on the label
55      * @param width  The prefered width
56      * @param height The prefered height
57      */
58     public ImageLabel(ImageIcon icon, int width, int height) {
59         this(icon);
60         setPreferredSize(new Dimension(width, height));
61     }
62 
63     /**
64      * Paints the label scaling the image to the appropriate size
65      *
66      * @param graphics The graphics context
67      */
68     public void paintComponent(Graphics graphics) {
69         Image image = this.imageIcon.getImage();
70         ImageObserver observer = imageIcon.getImageObserver();
71         ((Graphics2D) graphics).setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
72         ((Graphics2D) graphics).setRenderingHint(RenderingHints.KEY_INTERPOLATION, RenderingHints.VALUE_INTERPOLATION_BILINEAR);
73         graphics.drawImage(image, 0, 0, getWidth(), getHeight(), 0, 0, image.getWidth(observer), image.getHeight(observer), observer);
74     }
75 }
