16 package com.blogofbug.swing.components;
17 
18 import java.awt.Image;
19 import java.io.File;
20 import java.net.URL;
21 
22 /**
23  * A component that can be part of the a rich user interface
24  *
25  * @author nigel
26  */
27 public interface RichComponent {
28 
29     /**
30      * Assigns an image to the component, the width and height taken from the supplied image
31      *
32      * @param image The URL of the image
33      */
34     public void setRichImage(URL image);
35 
36     /**
37      * Assigns an image to the component, the width and height taken from the supplied image
38      *
39      * @param image A file referencing the image
40      */
41     public void setRichImage(File image);
42 
43     /**
44      * Assisgns an image to the component, the width and height taken from the supplied image
45      *
46      * @param image An image object containing the image
47      */
48     public void setRichImage(Image image);
49 
50     /**
51      * Allows a RichContainer to request that the component pre-renders anything
52      * that might provide peak performance. It is expected that most implementations will
53      * do this automatically after the image has been specified
54      */
55     public void prePaintImage();
56 
57     /**
58      * Specifies text that should be displayed by the rich component when
59      * getRichText() is called.
60      *
61      * @param text The text to be displayed. This may be different to that of the component
62      *             being enriched.
63      */
64     public void setRichText(String text);
65 
66     /**
67      * Gets the text to be displayed by the RichContainer for the component. May
68      * be extracted from elsewhere
69      *
70      * @return The text to be displayed
71      */
72     public String getRichText();
73 
74     /**
75      * Assisgns a uniform alpha to the component
76      *
77      * @param alpha A value from 0.0 to 1.0 where 1.0 is fully visible, and 0.0
78      *              is completely invisible.
79      */
80     public void setAlpha(float alpha);
81 
82 
83 }
