049     /**
050      * Creates an image compatible with the current display
051      *
052      * @return A BufferedImage with the appropriate color model
053      */
054     public static BufferedImage createCompatibleImage(int width, int height) {
055         GraphicsConfiguration configuration = GraphicsEnvironment.getLocalGraphicsEnvironment().getDefaultScreenDevice().getDefaultConfiguration();
056 
057         //Create a buffered image which is the right (translucent) format for the current graphics device, this
058         //should ensure the fastest possible performance. Adding on some extra height to make room for the reflection
059         return configuration.createCompatibleImage(width, height, Transparency.TRANSLUCENT);
060     }
061 
062     /**
063      * Loads an image in a format compatible with the current display
064      *
065      * @return A BufferedImage with the appropriate color model
066      */
067     public static BufferedImage loadCompatibleImage(String imageURL) {
068         Image image = null;
069         try {
070             image = ImageIO.read(new URL(imageURL));
071         } catch (MalformedURLException ex) {
072             ex.printStackTrace();
073             return null;
074         } catch (IOException ex) {
075             ex.printStackTrace();
076             return null;
077         }
078         if (image == null) {
079             return null;
080         }
081 
082         GraphicsConfiguration configuration = GraphicsEnvironment.getLocalGraphicsEnvironment().getDefaultScreenDevice().getDefaultConfiguration();
083 
084         //Create a buffered image which is the right (translucent) format for the current graphics device, this
085         //should ensure the fastest possible performance. Adding on some extra height to make room for the reflection
086         BufferedImage originalImage = configuration.createCompatibleImage(image.getWidth(null), image.getHeight(null), Transparency.TRANSLUCENT);
087 
088         //Blit the loaded image onto the optimized surface by creating a graphics context for the new image
089         Graphics2D g = originalImage.createGraphics();
090         //Draw the original image
091         g.drawImage(image, 0, 0, null);
092         g.dispose();
093         return originalImage;
094     }
095 
096     /**
097      * Produces a resized image that is of the given dimensions
098      *
099      * @param image  The original image
100      * @param width  The desired width
101      * @param height The desired height
102      * @return The new BufferedImage
103      */
104     public static BufferedImage scaledImage(BufferedImage image, int width, int height) {
105         BufferedImage newImage = createCompatibleImage(width, height);
106         Graphics graphics = newImage.createGraphics();
107 
108         graphics.drawImage(image, 0, 0, width, height, null);
109 
110         graphics.dispose();
111         return newImage;
112     }

114     /**
115      * Produces a copy of the supplied image
116      *
117      * @param image The original image
118      * @return The new BufferedImage
119      */
120     public static BufferedImage copyImage(BufferedImage image) {
121         return scaledImage(image, image.getWidth(), image.getHeight());
122     }

124     /**
125      * Renders a paragraph of text (line breaks ignored) to an image (created and returned).
126      *
127      * @param font      The font to use
128      * @param textColor The color of the text
129      * @param text      The message
130      * @param width     The width the text should be limited to
131      * @return An image with the text rendered into it
132      */
133     public static BufferedImage renderTextToImage(Font font, Color textColor, String text, int width) {
134         Hashtable map = new Hashtable();
135         map.put(TextAttribute.FONT, font);
136         AttributedString attributedString = new AttributedString(text, map);
137         AttributedCharacterIterator paragraph = attributedString.getIterator();
138 
139         FontRenderContext frc = new FontRenderContext(null, false, false);
140         int paragraphStart = paragraph.getBeginIndex();
141         int paragraphEnd = paragraph.getEndIndex();
142         LineBreakMeasurer lineMeasurer = new LineBreakMeasurer(paragraph, frc);
143 
144         float drawPosY = 0;
145 
146         //First time around, just determine the height
147         while (lineMeasurer.getPosition() < paragraphEnd) {
148             TextLayout layout = lineMeasurer.nextLayout(width);
149 
150             // Move it down
151             drawPosY += layout.getAscent() + layout.getDescent() + layout.getLeading();
152         }
153 
154         BufferedImage image = ImageUtilities.createCompatibleImage(width, (int) drawPosY);
155         Graphics2D graphics = (Graphics2D) image.getGraphics();
156         graphics.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING, RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
157 
158         drawPosY = 0;
159         lineMeasurer.setPosition(paragraphStart);
160         while (lineMeasurer.getPosition() < paragraphEnd) {
161             TextLayout layout = lineMeasurer.nextLayout(width);
162 
163             // Move y-coordinate by the ascent of the layout.
164             drawPosY += layout.getAscent();
165             
166            /* Compute pen x position.  If the paragraph is
167               right-to-left, we want to align the TextLayouts
168               to the right edge of the panel.
169             */
170             float drawPosX;
171             if (layout.isLeftToRight()) {
172                 drawPosX = 0;
173             } else {
174                 drawPosX = width - layout.getAdvance();
175             }
176 
177             // Draw the TextLayout at (drawPosX, drawPosY).
178             layout.draw(graphics, drawPosX, drawPosY);
179 
180             // Move y-coordinate in preparation for next layout.
181             drawPosY += layout.getDescent() + layout.getLeading();
182         }
183 
184         graphics.dispose();
185         return image;
186     }

189     /**
190      * Renders multiple paragraphs of text in an array to an image (created and returned).
191      *
192      * @param font      The font to use
193      * @param textColor The color of the text
194      * @param text      The message in an array of strings (one paragraph in each
195      * @param width     The width the text should be limited to
196      * @return An image with the text rendered into it
197      */
198     public static BufferedImage renderTextToImage(Font font, Color textColor, String text[], int width) {
199         LinkedList<BufferedImage> images = new LinkedList<BufferedImage>();
200 
201         int totalHeight = 0;
202 
203         for (String paragraph : text) {
204             BufferedImage paraImage = renderTextToImage(font, textColor, paragraph, width);
205             totalHeight += paraImage.getHeight();
206             images.add(paraImage);
207         }
208 
209         BufferedImage image = createCompatibleImage(width, totalHeight);
210         Graphics2D graphics = (Graphics2D) image.createGraphics();
211 
212         int y = 0;
213 
214         for (BufferedImage paraImage : images) {
215             graphics.drawImage(paraImage, 0, y, null);
216             y += paraImage.getHeight();
217         }
218 
219         graphics.dispose();
220         return image;
221     }

223     /**
224      * Renders a component into an image, which is useful for playing with the component's
225      * resultant image in special effects or transitions
226      *
227      * @param component The component to render
228      * @return A buffered image with the rendered component.
229      */
230     public static BufferedImage renderComponentToImage(JComponent component) {
231         //Create the image
232         BufferedImage image = createCompatibleImage(component.getWidth(), component.getHeight());
233 
234         //Render the component onto the image
235         Graphics graphics = image.createGraphics();
236         component.paint(graphics);
237         graphics.dispose();
238         return image;
239     }
